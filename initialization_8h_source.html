151 template <int N>
152 Vector<Offset,N> computeStrides(Vector<Size,N> const & shape, DataOrderEnum order=ROW_MAJOR) {
153  Vector<Offset,N> r(1);
154  if (order == ROW_MAJOR) {
155  for (int n=N-1; n > 0; --n) r[n-1] = r[n] * shape[n];
156  } else {
157  for (int n=1; n < N; ++n) r[n] = r[n-1] * shape[n-1];
158  }
159  return r;
160 }
